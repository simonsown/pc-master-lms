// Path: middleware.ts
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'

const ROUTE_PERMISSIONS: Record<string, string[]> = {
  '/admin':        ['admin'],
  '/teacher':      ['teacher', 'admin'],
  '/student':      ['student', 'teacher', 'admin'],
  '/parent':       ['parent', 'admin'],
  '/builder':      ['student', 'teacher', 'admin'],
  '/profile':      ['student', 'teacher', 'admin', 'parent'],
  '/notifications':['student', 'teacher', 'admin', 'parent'],
  '/leaderboard':  ['student', 'teacher', 'admin', 'parent'],
}

const PUBLIC = ['/', '/login', '/register', '/about',
  '/forgot-password', '/reset-password', '/verify']

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl
  let res = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll: () => request.cookies.getAll(),
        setAll: (toSet) => {
          toSet.forEach(({ name, value }) => request.cookies.set(name, value))
          res = NextResponse.next({ request })
          toSet.forEach(({ name, value, options }) => res.cookies.set(name, value, options))
        },
      },
    }
  )

  // LUÔN dùng getUser() — không dùng getSession()
  const { data: { user } } = await supabase.auth.getUser()

  const isPublic = PUBLIC.some(r => pathname === r || pathname.startsWith(r + '/'))

  if (isPublic) {
    if (user && (pathname === '/login' || pathname === '/register')) {
      const { data: p } = await supabase
        .from('profiles').select('role').eq('id', user.id).single()
      const dash = { admin:'/admin/dashboard', teacher:'/teacher/dashboard',
        parent:'/parent/dashboard', student:'/student/dashboard' }
      return NextResponse.redirect(new URL(dash[p?.role as keyof typeof dash] ?? '/student/dashboard', request.url))
    }
    return res
  }

  if (!user) {
    const url = new URL('/login', request.url)
    url.searchParams.set('redirect', pathname)
    return NextResponse.redirect(url)
  }

  const { data: profile } = await supabase
    .from('profiles').select('role').eq('id', user.id).single()
  const role = profile?.role ?? 'student'

  const matched = Object.keys(ROUTE_PERMISSIONS).find(r => pathname.startsWith(r))
  if (matched && !ROUTE_PERMISSIONS[matched].includes(role)) {
    const dash = { admin:'/admin/dashboard', teacher:'/teacher/dashboard',
      parent:'/parent/dashboard', student:'/student/dashboard' }
    return NextResponse.redirect(new URL(dash[role as keyof typeof dash] ?? '/student/dashboard', request.url))
  }

  res.headers.set('x-user-role', role)
  res.headers.set('x-user-id', user.id)
  return res
}

// Dùng trong Server Components: lấy user hiện tại
export async function getCurrentUser(): Promise<AuthUser | null> {
  const cookieStore = await cookies()
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    { cookies: { getAll: () => cookieStore.getAll(), setAll: () => {} } }
  )
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return null

  const { data: profile } = await supabase
    .from('profiles')
    .select('full_name, role, student_code, avatar_url')
    .eq('id', user.id).single()

  if (!profile) return null
  return {
    id: user.id,
    email: user.email!,
    role: profile.role as UserRole,
    full_name: profile.full_name,
    student_code: profile.student_code ?? undefined,
    avatar_url: profile.avatar_url ?? undefined,
  }
}

// Guard Server Component — redirect nếu không đúng role
export async function requireRole(allowedRoles: UserRole[]): Promise<AuthUser> {
  const user = await getCurrentUser()
  if (!user) redirect('/login')
  if (!allowedRoles.includes(user.role)) {
    const dash: Record<UserRole, string> = {
      admin: '/admin/dashboard', teacher: '/teacher/dashboard',
      parent: '/parent/dashboard', student: '/student/dashboard',
    }
    redirect(dash[user.role])
  }
  return user
}

// Guard Server Action — throw error thay vì redirect
export async function verifyRole(allowedRoles: UserRole[]): Promise<AuthUser> {
  const user = await getCurrentUser()
  if (!user) throw new Error('UNAUTHORIZED')
  if (!allowedRoles.includes(user.role)) throw new Error('FORBIDDEN')
  return user
}

        setAll: (c) => c.forEach(({ name, value, options }) => cookieStore.set(name, value, options))
      }
    }
  )
}

CREATE OR REPLACE FUNCTION generate_student_code() RETURNS TEXT AS $$
DECLARE chars TEXT := 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'; result TEXT := 'PCM-'; i INT;
BEGIN
  FOR i IN 1..8 LOOP
    result := result || substr(chars, floor(random()*length(chars)+1)::int, 1);
  END LOOP; RETURN result;
END; $$ LANGUAGE plpgsql;

CREATE OR REPLACE FUNCTION auto_student_code() RETURNS TRIGGER AS $$
BEGIN
  IF NEW.role = 'student' AND NEW.student_code IS NULL THEN
    LOOP NEW.student_code := generate_student_code();
      EXIT WHEN NOT EXISTS(SELECT 1 FROM profiles WHERE student_code = NEW.student_code);
    END LOOP;
  END IF; RETURN NEW;
END; $$ LANGUAGE plpgsql;

CREATE TRIGGER trig_student_code BEFORE INSERT ON public.profiles
  FOR EACH ROW EXECUTE FUNCTION auto_student_code();

import { createServerActionSupabase } from '@/lib/auth/rbac'

export default async function StudentDashboardPage() {
  const user = await requireRole(['student', 'teacher', 'admin'])
  const supabase = await createServerActionSupabase()

  // Parallel fetch — không await tuần tự
  const [progressRes, quizRes, recentRes] = await Promise.all([
    supabase.from('lesson_progress')
      .select('status, time_spent_seconds, completed_at, last_accessed')
      .eq('student_id', user.id),
    supabase.from('quiz_attempts')
      .select('score, submitted_at, status')
      .eq('student_id', user.id)
      .eq('status', 'graded')
      .order('submitted_at', { ascending: false })
      .limit(10),
    supabase.from('lesson_progress')
      .select('lesson_id, status, completion_percentage, last_accessed')
      .eq('student_id', user.id)
      .eq('status', 'in_progress')
      .order('last_accessed', { ascending: false })
      .limit(1),
  ])

  const progress = progressRes.data ?? []
  const quizzes = quizRes.data ?? []

  // Tính toán stats server-side
  const stats = {
    completed: progress.filter(p => p.status === 'completed').length,
    total: progress.length,
    totalMinutes: Math.round(
      progress.reduce((s, p) => s + (p.time_spent_seconds ?? 0), 0) / 60
    ),
    avgScore: quizzes.length > 0
      ? Math.round(quizzes.reduce((s, q) => s + (q.score ?? 0), 0) / quizzes.length)
      : null,
  }

  // Chart data: 7 ngày
  const days = Array.from({length: 7}, (_, i) => {
    const d = new Date(); d.setDate(d.getDate() - (6 - i))
    return d.toISOString().split('T')[0]
  })

  const chartData = days.map(day => {
    const dayProgress = progress.filter(p =>
      p.last_accessed?.startsWith(day)
    )
    return {
      day: day.slice(5), // MM-DD
      minutes: Math.round(dayProgress.reduce((s, p) => s + (p.time_spent_seconds ?? 0), 0) / 60)
    }
  })

  return (
    <StudentDashboardClient
      user={user}
      stats={stats}
      chartData={chartData}
      inProgressLesson={recentRes.data?.[0] ?? null}
    />
  )
}

// Bắt đầu làm bài (học sinh)
export async function startQuizAttempt(quizId: string) {
  const user = await verifyRole(['student'])
  const supabase = await createServerActionSupabase()

  // 1. Kiểm tra quiz published + trong thời gian
  const { data: quiz } = await supabase
    .from('quizzes').select('*').eq('id', quizId).eq('is_published', true).single()
  if (!quiz) return { error: 'Quiz không tồn tại hoặc chưa mở' }

  // 2. Đếm số lần đã làm
  const { count } = await supabase.from('quiz_attempts')
    .select('*', { count: 'exact', head: true })
    .eq('student_id', user.id).eq('quiz_id', quizId)
    .neq('status', 'in_progress')
  if (quiz.max_attempts && (count ?? 0) >= quiz.max_attempts) {
    return { error: `Bạn đã hết ${quiz.max_attempts} lần thử` }
  }

  // 3. Tạo attempt
  const { data: attempt } = await supabase.from('quiz_attempts').insert({
    student_id: user.id, quiz_id: quizId,
    quiz_title: quiz.title,
    attempt_number: (count ?? 0) + 1
  }).select().single()

  // 4. Lấy câu hỏi (randomize nếu cần) qua view an toàn
  const { data: questions } = await supabase
    .from('quiz_questions_safe').select('*').eq('quiz_id', quizId)

  let orderedQuestions = questions ?? []
  if (quiz.randomize_questions) {
    // Fisher-Yates shuffle
    for (let i = orderedQuestions.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1))
      ;[orderedQuestions[i], orderedQuestions[j]] = [orderedQuestions[j], orderedQuestions[i]]
    }
  }

  if (quiz.randomize_options) {
    orderedQuestions = orderedQuestions.map(q => ({
      ...q, options: [...q.options].sort(() => Math.random() - 0.5)
    }))
  }

  return { attemptId: attempt?.id, questions: orderedQuestions, quiz }
}

// Nộp bài (học sinh) — chấm điểm server-side
export async function submitQuizAttempt(
  attemptId: string,
  answers: { questionId: string; selectedOptionIds?: string[]; textAnswer?: string }[]
) {
  const user = await verifyRole(['student'])
  const supabase = await createServerActionSupabase()

  // 1. Verify attempt thuộc về user này và đang in_progress
  const { data: attempt } = await supabase.from('quiz_attempts')
    .select('*').eq('id', attemptId).eq('student_id', user.id)
    .eq('status', 'in_progress').single()
  if (!attempt) return { error: 'Không thể nộp bài này' }

  // 2. Lấy đáp án đúng từ DB (server-side)
  const questionIds = answers.map(a => a.questionId)
  const { data: correctOptions } = await supabase.from('question_options')
    .select('id, question_id, is_correct')
    .in('question_id', questionIds)

  const { data: questions } = await supabase.from('questions')
    .select('id, points, type').in('id', questionIds)

  // 3. Chấm điểm từng câu
  let totalScore = 0, maxScore = 0
  const answerRecords = answers.map(answer => {
    const q = questions?.find(q => q.id === answer.questionId)
    const qPoints = q?.points ?? 10
    maxScore += qPoints
    const correctIds = correctOptions
      ?.filter(o => o.question_id === answer.questionId && o.is_correct)
      .map(o => o.id) ?? []

    let isCorrect = false
    if (q?.type === 'fill_blank') {
      isCorrect = (answer.textAnswer?.toLowerCase().trim() ===
        correctIds[0]?.toLowerCase().trim()) ?? false
    } else if (q?.type === 'single_choice' || q?.type === 'true_false') {
      isCorrect = answer.selectedOptionIds?.length === 1 &&
        answer.selectedOptionIds[0] === correctIds[0]
    } else if (q?.type === 'multiple_choice') {
      const selected = new Set(answer.selectedOptionIds ?? [])
      const correct = new Set(correctIds)
      isCorrect = selected.size === correct.size &&
        [...selected].every(id => correct.has(id))
    }

    if (isCorrect) totalScore += qPoints
    return {
      attempt_id: attemptId, question_id: answer.questionId,
      selected_option_ids: answer.selectedOptionIds,
      text_answer: answer.textAnswer,
      is_correct: isCorrect,
      points_earned: isCorrect ? qPoints : 0
    }
  })

  const scorePercent = maxScore > 0 ? Math.round(totalScore / maxScore * 100) : 0

  // 4. Lưu answers và update attempt
  await Promise.all([
    supabase.from('quiz_answers').insert(answerRecords),
    supabase.from('quiz_attempts').update({
      status: 'graded', submitted_at: new Date().toISOString(),
      score: scorePercent, max_score: 100
    }).eq('id', attemptId)
  ])

  return { score: scorePercent, passed: scorePercent >= (attempt.passing_score ?? 70), attemptId }
}

'use client'
import { useEffect, useState, useRef } from 'react'
import { createClient } from '@/lib/supabase-client'

export function useNotifications(userId: string | null) {
  const [notifications, setNotifications] = useState<any[]>([])
  const [unreadCount, setUnreadCount] = useState(0)
  const supabase = createClient()
  const channelRef = useRef<any>(null)

  useEffect(() => {
    if (!userId) return

    // Load ban đầu
    supabase.from('notifications').select('*')
      .eq('user_id', userId).order('created_at', { ascending: false }).limit(20)
      .then(({ data }) => {
        setNotifications(data ?? [])
        setUnreadCount(data?.filter(n => !n.is_read).length ?? 0)
      })

    // Realtime subscription
    channelRef.current = supabase.channel(`notif-${userId}`)
      .on('postgres_changes', {
        event: 'INSERT', schema: 'public', table: 'notifications',
        filter: `user_id=eq.${userId}`
      }, (payload) => {
        setNotifications(prev => [payload.new, ...prev])
        setUnreadCount(prev => prev + 1)
      })
      .subscribe()

    return () => {
      if (channelRef.current) supabase.removeChannel(channelRef.current)
    }
  }, [userId])

  const markAsRead = async (id: string) => {
    setNotifications(prev => prev.map(n => n.id===id ? {...n, is_read: true} : n))
    setUnreadCount(prev => Math.max(0, prev - 1))
    await supabase.from('notifications').update({ is_read: true }).eq('id', id)
  }

  const markAllAsRead = async () => {
    setNotifications(prev => prev.map(n => ({...n, is_read: true})))
    setUnreadCount(0)
    await supabase.from('notifications').update({ is_read: true })
      .eq('user_id', userId!).eq('is_read', false)
  }

  return { notifications, unreadCount, markAsRead, markAllAsRead }
}

DECLARE chars TEXT := 'ABCDEFGHJKLMNPQRSTUVWXYZ23456789'; result TEXT; i INT;
BEGIN
  result := 'PCM-' || EXTRACT(YEAR FROM NOW())::TEXT || '-';
  FOR i IN 1..6 LOOP result := result || substr(chars, floor(random()*length(chars)+1)::int, 1); END LOOP;
  RETURN result;

export function useChildRealtime(studentId: string | null): ChildRealtimeData {

    isOnline: false, lastSeen: null,

  const channelRef = useRef<any>(null)
  const supabase = createClient()

  const fetchInitial = useCallback(async () => {
    if (!studentId) return
    const [{ data: progress }, { data: quizzes }, { data: profile }] = await Promise.all([
      supabase.from('lesson_progress').select('status,time_spent_seconds,completed_at,last_accessed')
        .eq('student_id', studentId),
      supabase.from('quiz_attempts').select('score,submitted_at,quiz_title,status')
        .eq('student_id', studentId).eq('status','graded').order('submitted_at',{ascending:false}).limit(10),
      supabase.from('profiles').select('is_online,last_seen_at').eq('id', studentId).single(),
    ])

    const p = progress ?? []
    const q = quizzes ?? []
    const completed = p.filter(x => x.status==='completed')
    const totalSec = p.reduce((s,x) => s+(x.time_spent_seconds??0), 0)
    const avg = q.length > 0 ? q.reduce((s,x) => s+(x.score??0), 0)/q.length : null

    const activities: Activity[] = [
      ...p.filter(x => x.status==='completed' && x.completed_at)

    ].sort((a,b) => new Date(b.timestamp).getTime()-new Date(a.timestamp).getTime()).slice(0,8)

    setData(prev => ({
      ...prev,
      isOnline: profile?.is_online ?? false,
      lastSeen: profile?.last_seen_at ?? null,
      lessonsCompleted: completed.length,
      totalLessonsStarted: p.length,
      totalTimeSeconds: totalSec,
      avgQuizScore: avg ? Math.round(avg) : null,
      recentActivities: activities,
    }))

            setData(prev => ({
              ...prev,
              recentActivities: [{
                id:`q-${r.id}`, type:'quiz_graded',
                title:`Quiz: ${r.quiz_title}`, timestamp: new Date().toISOString(),
                score: r.score
              }, ...prev.recentActivities].slice(0,8)
            }))
          }
        })
      .on('postgres_changes', { event:'UPDATE', schema:'public',
        table:'profiles', filter:`id=eq.${studentId}` },
        (payload) => {
          const r = payload.new as any
          setData(prev => ({ ...prev, isOnline: r.is_online ?? false, lastSeen: r.last_seen_at }))
        })
      .subscribe((status) => {
        setData(prev => ({
          ...prev,
          connectionStatus: status==='SUBSCRIBED' ? 'connected'
            : status==='CHANNEL_ERROR' ? 'error'
            : status==='CLOSED' ? 'disconnected' : 'connecting'
        }))
      })

    return () => {
      if (channelRef.current) {
        supabase.removeChannel(channelRef.current)
        channelRef.current = null
      }
    }
  }, [studentId, fetchInitial])

  return data
}

// Gửi yêu cầu liên kết
export async function requestLinkAction(studentId: string, relationship: string) {
  const parent = await verifyRole(['parent', 'admin'])
  const supabase = await createServerActionSupabase()

  const { data: student } = await supabase.from('profiles')
    .select('id,role,full_name').eq('id', studentId).single()
  if (!student || student.role !== 'student') return { error: 'Học sinh không tồn tại' }

  const { error } = await supabase.from('parent_student_links').upsert({
    parent_id: parent.id, student_id: studentId, relationship, status: 'pending'
  }, { onConflict: 'parent_id,student_id' })
  if (error) return { error: 'Không thể gửi yêu cầu' }

  await supabase.from('notifications').insert({
    user_id: studentId, type: 'parent_link_request',
    title: '👨‍👩‍👧 Phụ huynh muốn theo dõi bạn',
    body: 'Có phụ huynh gửi yêu cầu liên kết để theo dõi tiến độ học của bạn.',
    data: { parent_id: parent.id },
    action_url: '/student/profile/privacy'
  })

  revalidatePath('/parent/dashboard')
  return { success: true }
}

  await supabase.from('notifications').insert({
    user_id: parentId, type:'parent_link_confirmed',
    title:'✅ Học sinh đã xác nhận liên kết',
